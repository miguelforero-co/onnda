```ts
import { open } from "@tauri-apps/plugin-dialog";
import { invoke } from "@tauri-apps/api/core";
import { listen } from "@tauri-apps/api/event";
let uploadState = $state("");   // "" | "decoding" | "transcribing" | error text
async function uploadAudio() {
  const path = await open({ multiple:false, filters:[{ name:"Audio", extensions:["wav","mp3","m4a"] }] });
  if (!path || Array.isArray(path)) return;
  uploadState = "decoding";
  try { await invoke("transcribe_file", { path }); }
  catch (e) { uploadState = "No se pudo transcribir el archivo. Formatos admitidos: WAV, MP3, M4A."; }
}
```
Subscribe to events on mount (and unsubscribe on destroy): `file-transcribe-progress` → set uploadState ("Decodificando…"/"Transcribiendo…"); `file-transcribe-done` → clear uploadState + `onRefresh()` (re-pull history so the new file entry shows); `file-transcribe-error` → set uploadState to the Spanish error copy.
